app.post("/api/permissions", function(req, res){
    var params = req.body.params;
    var str = params.split(' ');

    sql.connect(config, function (err) {
        if (err) console.log(err);
        else {
            var request = new sql.Request();
            request.input("username", sql.VarChar, str[0]);
            request.input("databasename", sql.VarChar, str[1]);
            request.execute("GetUsers", function (err, result) {
                if (err) {
                    console.log(err);
                    return res.status(201).send(err);
                } else {
                    var permissionsList = [];
                    if (result.recordset.length > 0) {
                        for (var i = 0; i < result.recordset.length; i++) {
                            var permission = {};
                            permission.UserName = result.recordset[i]["UserName"];
                            permission.DatabaseName = result.recordset[i]["DatabaseName"];
                            permission.TableName = result.recordset[i]["TableName"];
                            permission.ReadPerm = result.recordset[i]["ReadPerm"];
                            permission.WritePerm = result.recordset[i]["WritePerm"];
                            permission.AlterPerm = result.recordset[i]["AlterPerm"];
                            permission.FullAccessPerm = result.recordset[i]["FullAccessPerm"];
                            permissionsList.push(permission);
                        }
                    }
                    sql.close();
                    return res.status(200).send({ data: permissionsList });
                }
            });
        }
    });
});
